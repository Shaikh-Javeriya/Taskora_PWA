const { app, BrowserWindow } = require("electron");
const path = require("path");
const { spawn } = require("child_process");
const fs = require("fs");

let win, nextServer;

function createWindow() {
  win = new BrowserWindow({
    width: 1280,
    height: 800,
    webPreferences: {
      nodeIntegration: false,
      contextIsolation: true,
    },
    icon: path.join(app.getAppPath(), "icon.png"),
  });

  win.loadURL("http://localhost:3000");
  //win.webContents.openDevTools({ mode: "undocked" });
}

app.whenReady().then(() => {
  const baseDir = app.getAppPath();
  const serverPath = path.join(baseDir, ".next", "standalone", "server.js");

  // üîç 1. Verify that file exists before spawning
  if (!fs.existsSync(serverPath)) {
    console.error("‚ùå Could not find:", serverPath);
    win = new BrowserWindow({
      width: 600,
      height: 400,
      title: "Taskora Error",
    });
    win.loadURL(
      "data:text/html,<h2 style='color:red;text-align:center;'>‚ùå Could not find Next.js server.js</h2>"
    );
    return;
  }

  console.log("üü¢ Starting Next.js standalone server...");
  console.log("üìÇ Path:", serverPath);

  // üß† 2. Spawn Node.js directly
  nextServer = spawn(process.execPath, [serverPath], {
    cwd: path.dirname(serverPath),
    env: { ...process.env, PORT: "3000" },
    stdio: "pipe",
  });

  // ‚úÖ 3. Log all events for debugging
  nextServer.stdout.on("data", (data) => {
    const msg = data.toString();
    console.log(`[next] ${msg}`);
    if (msg.includes("started server") || msg.includes("Ready")) {
      createWindow();
    }
  });

  nextServer.stderr.on("data", (data) => {
    console.error(`[next-error] ${data}`);
  });

  nextServer.on("error", (err) => {
    console.error("‚ùå Failed to start Next.js server:", err);
  });

  nextServer.on("exit", (code) => {
    console.log(`‚öôÔ∏è Next.js server exited with code ${code}`);
  });
});

app.on("window-all-closed", () => {
  if (process.platform !== "darwin") app.quit();
});

app.on("quit", () => {
  if (nextServer) nextServer.kill();
});
